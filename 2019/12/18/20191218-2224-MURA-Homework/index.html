  • 某顶尖外卖平台数据分析师面试题。现有交易数据表user_goods_table如下：
    user_name 用户名
    goods_kind 用户订购的的外卖品类

  • 现在老板想知道每个用户购买的外卖品类偏好分布，并取出每个用户购买最多的外卖品类是哪个。

  • 输出要求如下：
    user_name 用户名
    goods_kind 该用户购买的最多外卖品类

--13. 在emp表中, 查询在该月月底前两天入职的员工信息.
select * from
(
    select 
    e.empno, e.ename,
    month(date_sub(last_day(now()), interval 2 day)) target_m, 
    day(date_sub(last_day(now()), interval 2 day)) target_d, 
    month(hiredate), day(hiredate),
    day(date_sub(last_day(now()), interval 2 day)) - day(hiredate) jug
    from emp e
)A
where A.jug <= 2

+-------+--------+----------+----------+-----------------+---------------+------+
| empno | ename  | target_m | target_d | month(hiredate) | day(hiredate) | jug  |
+-------+--------+----------+----------+-----------------+---------------+------+
|  7004 | LIU    |        1 |       29 |               1 |            28 |    1 |
|  7028 | SHIN   |        1 |       29 |               1 |            29 |    0 |
|  7654 | MARTIN |        1 |       29 |               9 |            28 |    1 |
+-------+--------+----------+----------+-----------------+---------------+------+
3 rows in set (0.00 sec)

--下面的为正常写法
select * from
(
    select e.*,
    timestampdiff(DAY, e.hiredate, date_format(date_sub(last_day(now()), interval 2 day), '%Y-%m-%d')) as jug 
    from emp e
)A
where A.jug <= 2
;
