启动Zookeeper
zkServer.sh start
zkServer.sh stop
zkServer.sh status

TipsZookeeper启停需在所有节点执行，可以自定义

(base) [root@master bin]# cat zkServer_start_all.sh
# zkServer_start_all.sh
zkServer.sh start
ssh slave1 <<EOF
    zkServer.sh start
    exit
EOF
ssh slave2 <<EOF
        zkServer.sh start
    exit
EOF
sleep 3s
zkServer.sh status
######

# zkServer_stop_all.sh

ssh slave1 <<EOF
    zkServer.sh stop
    exit
EOF
ssh slave2 <<EOF
        zkServer.sh stop
    exit
EOF

2. Topic

  • 创建第一个Topic，并查看
kafka-topics.sh --create --zookeeper localhost:2181 --replication-factor 1 --partitions 2 --topic PREDATA
# --replication-factor 2 // 复制两份
# --partitions 1 // 创建1个分区
# --topic // 主题为my-topic
# --zookeeper // 此处为为zookeeper监听的地址


kafka-topics.sh --list --zookeeper localhost:2181
# 查看Topic状态
kafka-topics.sh --describe --zookeeper localhost:2181 --topic PREDATA
  • 删除第一个Topic
kafka-topics.sh --zookeeper localhost:2181 --delete --topic test
  • 创建第一个Consumer
kafka-console-consumer.sh --zookeeper master:2181 --topic test --from-beginning
# Using the ConsoleConsumer with old consumer is deprecated and will be removed in a future major release. Consider using the new consumer by passing [bootstrap-server] instead of [zookeeper].

# --zookeeper参数在后续版本会被取消，这里使用下面的命令替代
kafka-console-consumer.sh --bootstrap-server 10.72.8.46:9092 --topic test --from-beginning

# 从最新的offset开始消费
kafka-console-consumer.sh --bootstrap-server 10.72.8.46:9092 10.72.8.30:9092 10.72.8.38:9092 --topic PREDATA

# 从指定的offset开始消费，且指定偏移量时，需指定partition
kafka-console-consumer.sh --bootstrap-server master:9092 --topic test --offset 523 --partition 0
  • 创建第一个Producer
kafka-console-producer.sh --broker-list master:9092 --topic test
  • 查看日志
# Kafka日志需要利用其自带的工具才能查看
../../bin/kafka-run-class.sh kafka.tools.DumpLogSegments --files xxx.index  --print-data-log
